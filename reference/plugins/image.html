from rows.plugins.plugin_ocr import TesseractBackend
from rows.plugins.plugin_pdf import pdf_table_lines, TextObject, YGroupsAlgorithm
from rows.plugins.utils import create_table
# 
class MyTesseractBackend(TesseractBackend):
# 
    def objects(self, *args, **kwargs):
        for page in super().objects(*args, **kwargs):
            started, result = False, []
            for obj in page:
                if not started:
                    if obj.text == "DESCONTOS":
                        started = True
                    continue
                result.append(obj)
            result.sort(key=lambda obj: (obj.y0, obj.x0))
            yield result
# 
class MyFakeTesseractBackend(TesseractBackend):
# 
    def objects(self, *args, **kwargs):
        import json
        with open("objs.json") as fobj:
            r = []
            for obj in json.load(fobj):
                obj = TextObject(**obj)
                if len(obj.text) > 2:
                    r.append(obj)
            yield r

def import_from_image(
    filename_or_fobj,
    algorithm="y-groups",
    x_threshold=0.5,
    y_threshold=0.5,
    *args,
    **kwargs
):
    meta = {"imported_from": "image"}
    table_rows = pdf_table_lines(
        filename_or_fobj,
        None,
        starts_after=None,
        ends_before=None,
        algorithm=algorithm,
        x_threshold=x_threshold,
        y_threshold=y_threshold,
        backend=MyTesseractBackend,  # TODO: change
        backend_kwargs={"language": "por"},
    )
    return create_table(table_rows, meta=meta, *args, **kwargs)

def draw(backend, output_filename, background_image=None):
    from PIL import Image, ImageDraw, ImageFont

    pages = list(backend.objects())
#

TODO: plot all pages

    max_x = max(obj.x1 for obj in pages[0])
    max_y = max(obj.y1 for obj in pages[0])
    if background_image is None:
        image = Image.new("RGB", (max_x, max_y))
    else:
        image = Image.open(background_image)
    draw = ImageDraw.Draw(image)
    font = ImageFont.truetype(
        "/usr/share/fonts/truetype/dejavu/DejaVuSansMono.ttf", 30
    )

    for page in pages:
        for obj in page:
            text, x0, x1, y0, y1 = obj.text, obj.x0, obj.x1, obj.y0, obj.y1
            draw.rectangle(((x0, y0), (x1, y1)), outline="#FF0000")
            draw.rectangle(((x0+1, y0+1), (x1-1, y1-1)), outline="#FF0000")
            draw.rectangle(((x0+2, y0+2), (x1-2, y1-2)), outline="#FF0000")
            draw.text((x0, y0), text, font=font, fill="#FF0000")

    image.save(output_filename, "PNG")
