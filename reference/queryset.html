@dataclass
class Expression:
    field_name = None
    operation = None
    value = None


class Not(Expression):

    @classmethod
    def execute(self, op1):
        return not bool(op1)


class And(Expression):

    @classmethod
    def execute(self, op1, op2):
        return bool(op1) and bool(op2)


class Or(Expression):

    @classmethod
    def execute(self, op1, op2):
        return bool(op1) or bool(op2)
